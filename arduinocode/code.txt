static void sendReply(const char* s) {
  if (!txChar) return;
  txChar->setValue((uint8_t*)s, strlen(s));
  txChar->notify();
}

static String macToString(const byte mac[6]) {
  char buf[18];
  snprintf(buf, sizeof(buf), "%02X:%02X:%02X:%02X:%02X:%02X",
           mac[0], mac[1], mac[2], mac[3], mac[4], mac[5]);
  return String(buf);
}

// Reads status line + headers, returns HTTP status code, and leaves client positioned at body.
static int readHttpStatusAndSkipHeaders(EthernetClient &client, uint32_t timeoutMs) {
  uint32_t start = millis();

  // Read the status line: "HTTP/1.1 200 OK"
  String statusLine;
  while (millis() - start < timeoutMs) {
    while (client.available()) {
      char c = (char)client.read();
      if (c == '\r') continue;
      if (c == '\n') goto got_status;
      statusLine += c;
      if (statusLine.length() > 200) goto got_status; // safety
    }
    if (!client.connected()) break;
    delay(1);
  }
got_status:

  int code = -1;
  int sp1 = statusLine.indexOf(' ');
  if (sp1 >= 0) {
    int sp2 = statusLine.indexOf(' ', sp1 + 1);
    if (sp2 < 0) sp2 = statusLine.length();
    code = statusLine.substring(sp1 + 1, sp2).toInt();
  }

  // Skip headers until blank line
  String line;
  while (millis() - start < timeoutMs) {
    while (client.available()) {
      char c = (char)client.read();
      if (c == '\r') continue;
      if (c == '\n') {
        if (line.length() == 0) return code; // end of headers
        line = "";
      } else {
        line += c;
        if (line.length() > 300) line.remove(0, 200); // prevent runaway
      }
    }
    if (!client.connected()) break;
    delay(1);
  }

  return code;
}

// serverHost: e.g. "192.168.0.5" or "myserver.local"
// serverPort: usually 8080
// path: usually "/store"
bool sendMacAndIpToServer(const char *serverHost, uint16_t serverPort = 8080, const char *path = "/store") {
  EthernetClient client;

  // Build URL query
  String macStr = macToString(ethMac);

  // Basic URL encoding for ':' just in case (optional, but safe)
  String macEnc = macStr;
  macEnc.replace(":", "%3A");

  String reqPath = String(path) + "?mac=" + macEnc + "&ip=" + localIP;

  if (!client.connect(serverHost, serverPort)) {
    client.stop();
    return false;
  }

  // Send request
  client.print("GET ");
  client.print(reqPath);
  client.print(" HTTP/1.1\r\nHost: ");
  client.print(serverHost);
  client.print("\r\nConnection: close\r\n\r\n");

  // Read response: require HTTP 200 and body exactly "1"
  const uint32_t TIMEOUT_MS = 3000;
  int code = readHttpStatusAndSkipHeaders(client, TIMEOUT_MS);
  if (code != 200) {
    client.stop();
    return false;
  }

  // Read body (trim whitespace) and check it equals "1"
  uint32_t start = millis();
  String body;
  while (millis() - start < TIMEOUT_MS) {
    while (client.available()) {
      char c = (char)client.read();
      body += c;
      if (body.length() > 64) { // body should be tiny; prevent memory blowup
        // keep last 64 chars
        body.remove(0, body.length() - 64);
      }
    }
    if (!client.connected()) break;
    delay(1);
  }
  client.stop();

  body.trim();
  return (body == "1");
}

void generateRandomMac(byte *mac) {
  mac[0] = 0x14; mac[1] = 0x88;
  for (int i = 2; i < 6; i++) {
    mac[i] = esp_random() & 0xFF;
  }
}

void printMac(const byte *mac) {
  for (int i = 0; i < 6; i++) {
    if (i > 0) Serial.print(":");
    if (mac[i] < 16) Serial.print("0");
    Serial.print(mac[i], HEX);
  }
  Serial.println();
}

void read_settings() {
  Preferences pref;
  pref.begin("main", false);
  
  if (pref.isKey("ethMac")) {
    // Load saved MAC
    pref.getBytes("ethMac", ethMac, 6);
    ethMac[0] = 0x14; ethMac[1] = 0x88;
    Serial.print("Device MAC: ");
  } else {
    // Generate new random MAC
    generateRandomMac(ethMac);
    pref.putBytes("ethMac", ethMac, 6);
    Serial.print("Device new MAC: ");
  }
  printMac(ethMac);

  if (!pref.isKey("localIP") && !master) {
    Serial.println("Swap IP");
    // swap localIP / targetIP for slave device
    String tmpTargetIP = targetIP;
    targetIP = localIP;
    localIP = tmpTargetIP;
  }

  if (master) {
    dhcp_server = pref.getBool("dhcp", dhcp_server);
  }

  

  localIP = pref.getString("localIP", localIP);
  targetIP = pref.getString("targetIP", targetIP);
  addLog("LocalIP: "+localIP, "info");
  addLog("TargetIP: "+targetIP, "info");

  batt_cof = pref.getFloat("batt_cof", batt_cof);

  isBatteryConnected = pref.getBool("isBattery", isBatteryConnected);

  if(batt_cof < 0.5){
    batt_cof = 1.0f;
  }

  if(localIP == defaultIP){
    isNoIP = true;
  }

  crsfSpeed = pref.getULong("crsfSpeed", crsfSpeed);
  if (!master) {
    crsf2Speed = pref.getULong("crsf2Speed", crsf2Speed);
  }

  stringToIPAddress(targetIP+"", targetIPobj);

  pref.end();
}

void eth_setup() {
  ethOk = eth_init();
  
  if (master && dhcp_server) {
    addLog("DHCP server ON", "info");

    // reserve first 10 IPs for devices whose MAC starts 0x12,0x34
    dhcp.setReservedPrefix(0x14,0x88, 10);

    // turn on debug prints
    dhcp.setDebug(true);

    IPAddress ip = Ethernet.localIP();
    dhcp.begin(
      IPAddress(ip[0],ip[1],ip[2],10),
      IPAddress(ip[0],ip[1],ip[2],250),
      IPAddress(ip[0],ip[1],ip[2],1),
      IPAddress(255,255,255,0),
      UINT32_MAX
    );
  }

  if(ethOk){
    addLog("ETH Setup done", "ok");
  }else{
    addLog("ETH Setup error", "err");
  }
}

bool stringToIPAddress(const String& str, IPAddress& ip) {
  int parts[4] = {0};
  int index = 0;

  // Split the string by '.' and convert to integers
  char* token = strtok(const_cast<char*>(str.c_str()), ".");
  while (token != nullptr && index < 4) {
    parts[index++] = atoi(token);
    token = strtok(nullptr, ".");
  }

  // Check if we got exactly 4 parts
  if (index != 4) {
    return false; // Invalid IP address
  }

  // Validate each part (0-255) and assign to IPAddress
  for (int i = 0; i < 4; i++) {
    if (parts[i] < 0 || parts[i] > 255) {
      return false; // Invalid range
    }
  }

  ip = IPAddress(parts[0], parts[1], parts[2], parts[3]);
  return true;
}

bool eth_init() {
  stringToIPAddress(localIP+"", ethIp);
  //stringToIPAddress(cbIP+"", broadcastIp);

  if (master && dhcp_server) {
    ethIp[3] = 1;
    localIP = ethIp.toString();
  }

  // start the Ethernet connection and the server:
  Ethernet.begin(ethMac, ethIp);

  UdpCom.begin(COMM_PORT);
  UdpDisc.begin(DISCOVER_PORT);

  // Check for Ethernet hardware present
  if (Ethernet.hardwareStatus() == EthernetNoHardware) {
    addLog("Ethernet shield was not found.", "err");
    return false;
  }
  if (Ethernet.linkStatus() == LinkOFF) {
    addLog("Ethernet cable is not connected.", "err");
    isNoEthernet = true;
  }else{
    isNoEthernet = false;
  }

  // start the server
  ethServer.begin();
  // String etIP = String(Ethernet.localIP());
  String etIP = Ethernet.localIP().toString();
  addLog("Server is at " + etIP, "info");

  return true;
}

    writeInt(client, "UGMIN", UGN_MIN);
    writeInt(client, "UGMAX", UGN_MAX);
    writeInt(client, "UVMIN", UVN_MIN);
    writeInt(client, "UVMAX", UVN_MAX);
    client.print("servo_add=0\n");

    // AIM counter
    client.print("aim=0\naimc=0\naime=0\n");

    client.print(crsf_fail ? "CRSF=0\n" : "CRSF=1\n");
    
    client.print("RT=0\n");
    
    unsigned long currentMillis = millis();
    client.print("T=");
    client.print(currentMillis);
    client.print("\n");
    client.print("D=");
    client.print(currentMillis-now);
    client.print("\n");

    client.println();

    return true;
  }

  // set angles
  if (strncmp(page, "/acs/", 5) == 0) {
    getOkHead(client);

    int16_t numbers[4]; // Array to store the parsed integers
  
    char buffer[100];  // Make sure this is large enough to hold the string
    strncpy(buffer, page, sizeof(buffer) - 1); // Copy string to buffer
    buffer[sizeof(buffer) - 1] = '\0';         // Null-terminate just in case

    char *numberStart = strstr(buffer, "/acs/");
    if (numberStart != NULL) {
      numberStart += 5;  // Move pointer to start at "123.32.50.100.150"
      
      // Parse each number separated by '.'
      char *token = strtok(numberStart, ".");
      int i = 0;
      for (; i < 4 && token != NULL; i++) {
        numbers[i] = atoi(token);      // Convert token to integer
        token = strtok(NULL, ".");     // Move to the next token
      }
      if (i == 4) {
        double AX1_ = ((double) numbers[0]) / 100.0;
        double AY1_ = ((double) numbers[1]) / 100.0;
        double AX2_ = ((double) numbers[2]) / 100.0;
        double AY2_ = ((double) numbers[3]) / 100.0;

        double AX2_moff_new = AX2_ - AX2_value_raw;
        double AY2_moff_new = AY2_ - AY2_value_raw;

        AX2_moff = AX2_moff_new;
        AY2_moff = AY2_moff_new;
        
        client.print('1');
        client.print("\n");

        return true;
      }
    }

    client.print('0');
    client.print("\n");

    return true;
  }
  
  return false;
}

void eth_loop() {
  // listen for incoming clients
  EthernetClient client = ethServer.available();
  if (client) {
    //Serial.println("new client");
    // an HTTP request ends with a blank line
    bool currentLineIsBlank = false;
    bool end = false;

    int ethBuffPos = 0;
    bool ethPageFound = false;

    int space1pos = -1;
    int space2pos = -1;

    while (client.connected()) {
      if (client.available()) {
        char c = client.read();
        //Serial.write(c);
        // if you've gotten to the end of the line (received a newline
        // character) and the line is blank, the HTTP request has ended,
        // so you can send a reply
        if (c == ' ') {
          if (space1pos == -1)
            space1pos = ethBuffPos;
          else if (space2pos == -1)
            space2pos = ethBuffPos;
        }
        if (c != '\r' && c != '\n')
          ethBuff[ethBuffPos++] = c;
        if (c == '\n' && ethBuffPos > 0) {
          ethBuff[ethBuffPos] = '\0';
          
          if (!ethPageFound && space1pos > 0 && space2pos > 0) {
            memcpy(ethTmp, ethBuff, space1pos);
            ethTmp[space1pos] = '\0';
            if (strcmp(ethTmp, "GET") == 0) {
              int len = space2pos-space1pos-1;
              memcpy(ethPage, ethBuff+space1pos+1, len);
              ethPage[len] = '\0';
              ethPageFound = true;
            }
          }

          ethBuffPos = 0;
          space1pos = -1;
          space2pos = -1;
        }
        if (c == '\n' && currentLineIsBlank) {
          end = true;
          break;
        }
        if (c == '\n') {
          // you're starting a new line
          currentLineIsBlank = true;
        } else if (c != '\r') {
          // you've gotten a character on the current line
          currentLineIsBlank = false;
        }
      }
    }

    if (end) {
      bool ok = getPage(ethPage, client);
      if (!ok) {
        client.println("HTTP/1.1 404 not found");
        client.println();
        client.println();
      }
    }

    // close the connection:
    client.stop();
    if (ethPageFound) {
      //Serial.print("client finish (");
      //Serial.print(ethPage);
      //Serial.println(")");
    } else {
      //Serial.println("client finish (unknown page)");
    }
  }
  udp_discover_loop();
  udp_comm_loop();
  if (master && dhcp_server) {
    dhcp.handle();
  }
}

void handlePostParams() {
  String lIP  = server.arg("localIP");
  String tIP  = server.arg("targetIP");
  String spd  = server.arg("crsfSpeed");

  IPAddress ip;
  if(!ip.fromString(lIP)) {
    server.send(400, "text/plain", "Invalid local IP");
    return;
  }
  if(!ip.fromString(tIP)) {
    server.send(400, "text/plain", "Invalid target IP");
    return;
  }
  uint32_t baud = spd.toInt();
  if(baud == 0) {
    server.send(400, "text/plain", "Invalid CRSF baudrate");
    return;
  }

  if (master) {
    dhcp_server = server.hasArg("dhcpServer") && server.arg("dhcpServer").length() > 0;
  } else {
    String spd2  = server.arg("crsf2Speed");
    uint32_t baud2 = spd2.toInt();
    if(baud2 == 0) {
      server.send(400, "text/plain", "Invalid CRSF2 baudrate");
      return;
    }
    crsf2Speed = baud2;
  }

  localIP   = lIP;
  targetIP  = tIP;
  crsfSpeed = baud;
  write_settings();

  server.send(200, "text/plain", "OK");
  delay(1000);
  ESP.restart();
}

// POST /dhcp  →  kick off DHCP request
void handleDHCPStart() {
  if (!dhcpRequested) {
    // This will send the DHCPDISCOVER to your W5500
    Ethernet.begin(ethMac);
    dhcpRequested  = true;
    dhcpDone       = false;
    dhcpStartTime  = millis();
    dhcpIP         = IPAddress(0,0,0,0);
  }
  server.send(200, "text/plain", "OK");
}

// GET /dhcp/status  →  report back pending / ok / error
void handleDHCPStatus() {
  if (!dhcpRequested) {
    server.send(400, "text/plain", "Not started");
    return;
  }

  // once DHCP client has a lease, Ethernet.localIP() will be non‐zero
  if (!dhcpDone) {
    IPAddress got = Ethernet.localIP();
    if (got != IPAddress(0,0,0,0)) {
      dhcpIP    = got;
      dhcpDone  = true;
    }
  }

  String json;
  if (dhcpDone) {
    json = "{\"status\":\"ok\",\"ip\":\"" + dhcpIP.toString() + "\"}";
  }
  else if (millis() - dhcpStartTime > 10000) {
    // timeout after 10 s
    json = "{\"status\":\"error\"}";
  }
  else {
    json = "{\"status\":\"pending\"}";
  }

  server.send(200, "application/json", json);
}

// call when user clicks “Discover”
void handleDiscoverStart() {
  if (!discoverRequested) {
    deviceCount       = 0;
    discoverRequested = true;
    discoverStartTime = millis();
    UdpDisc.beginPacket("255.255.255.255", DISCOVER_PORT);
    UdpDisc.write(0x67);
    UdpDisc.endPacket();
  }
  server.send(200, "text/plain", "OK");
}

void send_v1_reply(IPAddress remoteIP) {
  IPAddress ip = Ethernet.localIP();
  Device dev;
  dev.mac[0] = ethMac[0]; dev.mac[1] = ethMac[1]; dev.mac[2] = ethMac[2]; dev.mac[3] = ethMac[3]; dev.mac[4] = ethMac[4]; dev.mac[5] = ethMac[5];
  dev.ip[0] = ip[0]; dev.ip[1] = ip[1]; dev.ip[2] = ip[2]; dev.ip[3] = ip[3];
  dev.master = master;
  UdpDisc.beginPacket(remoteIP, DISCOVER_PORT);
  UdpDisc.write(0x68);
  UdpDisc.write(reinterpret_cast<uint8_t*>(&dev), sizeof(Device));
  UdpDisc.endPacket();
}

void send_v2_reply(IPAddress remoteIP) {
  IPAddress ip = Ethernet.localIP();
  Device2 dev;
  dev.mac[0] = ethMac[0]; dev.mac[1] = ethMac[1]; dev.mac[2] = ethMac[2]; dev.mac[3] = ethMac[3]; dev.mac[4] = ethMac[4]; dev.mac[5] = ethMac[5];
  dev.ip[0] = ip[0]; dev.ip[1] = ip[1]; dev.ip[2] = ip[2]; dev.ip[3] = ip[3];
  dev.targetIp[0] = targetIPobj[0]; dev.targetIp[1] = targetIPobj[1]; dev.targetIp[2] = targetIPobj[2]; dev.targetIp[3] = targetIPobj[3];
  dev.master = master;
  dev.crsfSpeed = crsfSpeed;
  dev.crsf2Speed = master ? 0 : crsf2Speed;
  UdpDisc.beginPacket(remoteIP, DISCOVER_PORT);
  UdpDisc.write(0x71);
  UdpDisc.write(reinterpret_cast<uint8_t*>(&dev), sizeof(Device2));
  UdpDisc.endPacket();
}

void udp_comm_loop() {
  uint32_t now = millis();
  uint16_t timeout = master ? 5000 : 2000;
  if (!lanTimeout && now - lastLanEventTime > timeout) {
    lanTimeout = true;
    addLog("Connection with MASTER lost!", "err");
    d.isMaster = false;
  }

  int packetSize = UdpCom.parsePacket();
  if (packetSize == 0) return;
  
  char buf[64];
  int len = UdpCom.read(buf, sizeof(buf) - 1);
  if (len == 0) return;

  if (buf[0] == 0x69) {

    in_last_msg = now;
    lastLanEventTime = now;
    if (lanTimeout) {
      lanTimeout = false;
      addLog("Connection with MASTER restored!", "ok");
      d.isMaster = true;
    }
    crsf_port.write(buf+1, len-1);
    if (!master) crsf_last_msg = millis();
  } else if (buf[0] == 0x6A) {
    // heartbeat
    in_last_msg = now;
  } else if (buf[0] == 0x6B) {
    // set TARGET IP
    in_last_msg = now;
    targetIPobj = UdpCom.remoteIP();
    targetIP = targetIPobj.toString();
    write_settings();
    addLog("New target: " + targetIP, "ok");
    isPair = true;
  } else if (buf[0] == 0x6C) {
    // set LOCAL IP + crsf_speed + crsf2_speed
    in_last_msg = now;
    IPAddress localIPobj = IPAddress(buf[1], buf[2], buf[3], buf[4]);
    String oldIp = localIP;
    localIP = localIPobj.toString();
    uint32_t oldCrsfSpeed = crsfSpeed;
    uint32_t oldCrsf2Speed = crsf2Speed;

    uint32_t speed;
    memcpy(&speed, &buf[5], sizeof(speed));
    if (speed > 0) crsfSpeed = speed;
    if (!master) {
      memcpy(&speed, &buf[9], sizeof(speed));
      if (speed > 0) crsf2Speed = speed;
    }
    write_settings();
    if (oldIp != localIP) {
      ethOk = eth_init();

      addLog("Ethernet initiliazing... "+ ethOk, "info");
      if (ethOk) {
        ip_last_try = 0;
        ip_timeout = 0;
        ip_sent = false;
      }
    }
    if (crsfSpeed != oldCrsfSpeed) {
      crsf_port.end();
      delay(10);
      setup_crsf();
    }
    if (!master && crsf2Speed != oldCrsf2Speed) {
      crsf2_port.end();
      delay(10);
      setup_crsf2();
    }
    addLog("New local IP: "+localIP+" CRSF: "+crsfSpeed+" CRSF2: "+crsf2Speed, "ok");
    send_v2_reply(UdpCom.remoteIP());
  }
}

// non-blocking UDP handler: call this every loop()
void udp_discover_loop() {
  int packetSize = UdpDisc.parsePacket();
  if (packetSize == 0) return;

  IPAddress remoteIP = UdpDisc.remoteIP();
  char buf[64];
  int len = UdpDisc.read(buf, sizeof(buf) - 1);
  if (len == 0) return;
  
  // 1) If it’s a DISCOVER broadcast → reply with our MAC+master
  if (buf[0] == 0x67) {
    addLog("Received discover request! RemoteIp: "+remoteIP.toString(), "info");
    send_v1_reply(remoteIP);
    isFound = true;
    return;
  }

  // 1) If it’s a DISCOVERv2 broadcast → reply with our MAC+master
  if (buf[0] == 0x70) {
    //String remoteIPstr = ;
    addLog("Received discover request v2! RemoteIp: "+remoteIP.toString(), "info");
    send_v2_reply(remoteIP);
    isFound = true;
    return;
  }

  // 2) Otherwise treat it as a discovery-reply from others
  if (buf[0] == 0x68 && discoverRequested && millis() - discoverStartTime < 10000) {
    Device *d = (Device *)&buf[1];
    if (d->master != master) {
      devices[deviceCount++] = *d;
    }
  }
}

// returns JSON list of replies
void handleDiscoverResults() {
  // close window after 10 s
  if (discoverRequested && millis() - discoverStartTime >= 10000) {
    discoverRequested = false;
  }

  String js = "{\"devices\":[";
  for (int i = 0; i < deviceCount; i++) {
    char macSuffix[5];
    snprintf(macSuffix, sizeof(macSuffix), "%02X%02X", devices[i].mac[4], devices[i].mac[5]);
    String mac = String(macSuffix);

    char buf[16];  // max “255.255.255.255” + NUL
    snprintf(buf, sizeof(buf), "%u.%u.%u.%u", devices[i].ip[0], devices[i].ip[1], devices[i].ip[2], devices[i].ip[3]);
    String ipString = String(buf);

    js += "{\"mac\":\"" + mac + "\","
          "\"ip\":\""  + ipString  + "\","
          "\"master\":" + (devices[i].master ? "true" : "false") + "}";
    if (i < deviceCount - 1) js += ",";
  }
  js += "]}";
  server.send(200, "application/json", js);
}

  while(!SPIFFS.begin(true)) {
    Serial.println("SPIFFS Mount Failed!");
    delay(1000);
  }
  Serial.println("SPIFFS OK");

  String deviceName = String(AP_SSID);

  deviceName += master ? "_M" : "_S";

  deviceName += "_";
  deviceName += macSuffix;

  WiFi.softAPConfig(AP_IP, AP_GATEWAY, AP_SUBNET);
  WiFi.softAP(deviceName, AP_PASS);

  Serial.print("AP running @ "); Serial.print(WiFi.softAPIP());
  Serial.print(" (");
  Serial.print(deviceName);
  Serial.println(")");

  // static files
  server.serveStatic("/",        SPIFFS, "/index.html");
  server.serveStatic("/style.css",SPIFFS, "/style.css");
  server.serveStatic("/script.js",SPIFFS, "/script.js");

  // params API
  server.on("/params", HTTP_GET,  handleGetParams);
  server.on("/params", HTTP_POST, handlePostParams);

  server.on("/dhcp/status",  HTTP_GET,  handleDHCPStatus);

  server.on("/discover",         HTTP_POST, handleDiscoverStart);
  server.on("/discover/results", HTTP_GET,  handleDiscoverResults);

  server.onNotFound([](){
    server.send(404, "text/plain", "404: Not Found");
  });

  server.begin();
  Serial.println("Webserver started");
}

      gps_num_sat = gps.satellites.value();
      gps_deg = gps.course.deg();

        Serial.println(gps_lat, 6);
        Serial.print("Longitude: ");
        Serial.println(gps_lng, 6);
        Serial.print("Groundspeed: ");
        Serial.println(gps_speed, 1);
        Serial.print("Heading: ");
        Serial.println(gps_deg, 2);
        Serial.print("Altitude: ");
        Serial.println(gps_alt);
        Serial.print("Satellites: ");
        Serial.println(gps_num_sat);
        gps_timer = time;
      }
      #endif
    }
  }
  gps_ok = (gps.satellites.value() > 0) && (gps.location.age() < GPS_OK_AGE) ? 1 : 0;
  if(!gps_ok){
    isNoGPSSignal = true;
  }else{
    isNoGPSSignal = false;
  }
}

uint8_t crsf_crc8(const uint8_t *ptr, uint8_t len) {
  uint8_t crc = 0;
  for (uint8_t i = 0; i < len; i++) {
    crc = crsf_crc8tab[crc ^ *ptr++];
  }
  return crc;
}

void crsf_process_buffer() {
  uint32_t now = millis();

  switch (mtype) {

    // GPS
    case CRSF_FRAMETYPE_GPS:
      d.isNRK = true;
      isNRKTime = now;
      break;

    // Battery
    case CRSF_FRAMETYPE_BATTERY_SENSOR:
      if(isBatteryConnected){
        uint16_t v = batteryVoltage * 10.0f;

        inBuffer[3] = (v >> 8) & 0xFF;
        inBuffer[4] = v & 0xFF;

        inBuffer[bufferIndex - 1] = crsf_crc8(&inBuffer[2], inBuffer[1] - 1);
      }
      d.isNRK = true;
      isNRKTime = now;
      break;

    // RC channels from master
    case CRSF_FRAMETYPE_RC_CHANNELS_PACKED:
      d.isNRK = true;
      isNRKTime = now;
      break;

    default:
      if(now - isNRKTime > 1000){
        d.isNRK = false;
      }
    break;
  }

  if (master) crsf_last_msg = now;
  
  if (master || (!master && in_last_msg > 0)) {
    // відправляємо дані в лан якщо мастер-юніт або якщо слейв і отримано пакет від мастера.
    // якщо слейв спробує відправити пакет, коли мастер не був в мережі - будуть лаги
    UdpCom.beginPacket(targetIPobj, COMM_PORT);
    UdpCom.write(0x79);
    UdpCom.write(ethMac+2, 4);
    UdpCom.write(inBuffer, bufferIndex);
    UdpCom.endPacket();
    out_last_msg = now;
  }

  if (!master) {
    crsf2_port.write(inBuffer, bufferIndex);
  }
  
  /*
  if (addr == CRSF_ADDRESS_FLIGHT_CONTROLLER) {
    if (mtype == CRSF_FRAMETYPE_LINK_STATISTICS) {
      return;
    }
    if (mtype == CRSF_FRAMETYPE_RC_CHANNELS_PACKED) {
      return;
    }
  }*/
  
  /*
  Serial.print("FROM ");
  Serial.print(addr, HEX);
  Serial.print(" MSG ");
  Serial.println(mtype, HEX);
  */
}

void crsf_reset() {
  bufferIndex = 0;
  addr = 0;
  len = 0;
  mtype = 0;
  crc = 0;
  sync_ = false;
}

void loop_crsf() {
  if (!crsf_port.available()) {
    return;
  }

  uint8_t b = crsf_port.read();
  //Serial.println(b, HEX);
  
  if (!sync_) {
    if (b != CRSF_ADDRESS_BROADCAST && b != CRSF_ADDRESS_USB && b != CRSF_ADDRESS_TBS_CORE_PNP_PRO && b != CRSF_ADDRESS_RESERVED1 && b != CRSF_ADDRESS_CURRENT_SENSOR && b != CRSF_ADDRESS_GPS && b != CRSF_ADDRESS_TBS_BLACKBOX && b != CRSF_ADDRESS_FLIGHT_CONTROLLER && b != CRSF_ADDRESS_RESERVED2 && b != CRSF_ADDRESS_RACE_TAG && b != CRSF_ADDRESS_RADIO_TRANSMITTER && b != CRSF_ADDRESS_CRSF_RECEIVER && b != CRSF_ADDRESS_CRSF_TRANSMITTER) {
      return;
    }
    sync_ = true;
    bufferIndex = 0;
  }
  
  bool crc_ = false;
  if (bufferIndex == 0) {
    addr = b;
  } else if (bufferIndex == 1) {
    len = b;
    if (len < 3 || len > CRSF_MAX_PACKET_LEN) {
      crsf_reset();
      return;
    }
  } else if (bufferIndex == 2) {
    mtype = b;
  } else if (bufferIndex > 2 && bufferIndex == len+1) {
    crc = b;
    crc_ = true;
  }

  inBuffer[bufferIndex++] = b;
  if (bufferIndex >= CRSF_MAX_PACKET_LEN + 4) { // error?
    crsf_reset();
    return;
  }

  if (crc_) {
    uint8_t ncrc = crsf_crc8(&inBuffer[2], inBuffer[1] - 1);
    if (ncrc == crc) {
      crsf_process_buffer();
    } else {
      Serial.print("CRC ERROR ADDR=");
      Serial.print(addr, HEX);
      Serial.print(" MSG=");
      Serial.print(mtype, HEX);
      Serial.print("    CRC=");
      Serial.print(crc, HEX);
      Serial.print(" NCRC=");
      Serial.println(ncrc, HEX);
    }
    crsf_reset();
  }
}

// ----- BNO
float add_angles(float angle1, float angle2, float angle3) {
  // Add the angles
  float sum = angle1 + angle2 + angle3;

  // Normalize the result to the range -180 to +180 degrees
  while (sum > 180.0) sum -= 360.0;
  while (sum < -180.0) sum += 360.0;

  return sum;
}

float add_angles(float angle1, float angle2) {
  // Add the angles
  float sum = angle1 + angle2;

  // Normalize the result to the range -180 to +180 degrees
  while (sum > 180.0) sum -= 360.0;
  while (sum < -180.0) sum += 360.0;

  return sum;
}

// Function to compute the median of an array
float computeMedian(float *values, int size) {
  float sorted[size];
  memcpy(sorted, values, size * sizeof(float));  // Copy values to a temporary array
  for (int i = 0; i < size - 1; i++) {
    for (int j = i + 1; j < size; j++) {
      if (sorted[i] > sorted[j]) {
        float temp = sorted[i];
        sorted[i] = sorted[j];
        sorted[j] = temp;
      }
    }
  }
  // Return the middle value
  if (size % 2 == 0) {
    return (sorted[size / 2 - 1] + sorted[size / 2]) / 2.0;
  } else {
    return sorted[size / 2];
  }
}

void crsf_state_loop() {
  uint32_t now = millis();

  crsf1_fail = now - crsf_last_msg > 1000;
  crsf2_fail = now - crsf2_last_msg > 1000;

  crsf_fail = crsf1_fail;
  if (!master && crsf_fail) crsf_fail = crsf2_fail;
}

void send_heartbeat() {
  UdpCom.beginPacket(targetIPobj, COMM_PORT);
  UdpCom.write(0x6A);
  UdpCom.write(ethMac+2, 4);
  UdpCom.endPacket();
}

void loop_ip_www() {
  if (ip_sent) return;
  uint32_t now = millis();
  uint32_t msg_delay = now - ip_last_try;
  if (ip_timeout == 0) {
    ip_timeout = now + IP_DURATION_MIN*60000;
  }
  if (now >= ip_timeout) {
    addLog("TIMEOUT SENDING MAC+IP", "warn");
    ip_sent = true;
    return;
  }

  if (msg_delay >= IP_DELAY_MS) {
    bool ok = sendMacAndIpToServer(WWW_IP, 8080, "/store");
    // addLog(ok ? "MAC+IP SENT" : "ERROR SENDING MAC+IP", "info");
    if (ok) {
      ip_sent = true;
    }
    ip_last_try = now;
  }
}

void loop_heartbeat() {
  uint32_t now = millis();
  uint32_t msg_delay = now - out_last_hb;
  if (msg_delay >= 2000) {
    if (master) {
      send_heartbeat();
      out_last_hb = now;
      out_last_msg = now;
    } else {
      if (in_last_msg > 0) {
        send_heartbeat();
        out_last_hb = now;
        out_last_msg = now;
      }
    }
  }
}

void crsfQueuePacket(uint8_t addr, uint8_t type, const void *payload, uint8_t len) {
  if (len > CRSF_MAX_PACKET_LEN)
    return;
  
  uint32_t now = millis();

  uint8_t buf[CRSF_MAX_PACKET_LEN+4];
  buf[0] = addr;
  buf[1] = len + 2; // type + payload + crc
  buf[2] = type;
  memcpy(&buf[3], payload, len);
  //buf[len+3] = _crc.calc(&buf[2], len + 1);
  buf[len+3] = crsf_crc8(&buf[2], len + 1);
  
  //crsf1.write(buf, len + 4);
  if (!master && in_last_msg > 0) {
    // відправляємо дані в лан якщо мастер-юніт або якщо слейв і отримано пакет від мастера.
    // якщо слейв спробує відправити пакет, коли мастер не був в мережі - будуть лаги
    UdpCom.beginPacket(targetIPobj, COMM_PORT);
    UdpCom.write(0x79);
    UdpCom.write(ethMac+2, 4);
    UdpCom.write(buf, len + 4);
    UdpCom.endPacket();
    out_last_msg = now;
  }
}

uint32_t lastGpsTime;
void gps_telemetry_loop() {
  if (!gps_ok) return;

  // send GPS every 1sec
  uint32_t now = millis();
  if (now - lastGpsTime < 1000) return;
  lastGpsTime = now;
  
  crsf_sensor_gps_t gps = { 0 };

  gps.latitude = htobe32((int32_t)(gps_lat*10000000.0));
  gps.longitude = htobe32((int32_t)(gps_lng*10000000.0));
  gps.heading = htobe16((uint16_t)(gps_deg*100.0));
  gps.groundspeed = htobe16((uint16_t)(gps_speed*10.0));
  gps.altitude = htobe16((uint16_t)(gps_alt+1000.0));
  gps.satellites = (uint8_t)gps_num_sat;

  crsfQueuePacket(CRSF_SYNC_BYTE, CRSF_FRAMETYPE_GPS, &gps, sizeof(gps));
}

void loop() {
  eth_loop();
  web_loop();
  if (confMode) {
    server.handleClient();
  } else {
    loop_crsf();
    if (!master) loop_crsf2();
    loop_heartbeat();
    loop_ip_www();
  }
  // if (bno_ok) bno_loop();
  if (!no_gps) {
    gps_loop();
    gps_telemetry_loop();
  }
  crsf_state_loop();


  #if HW_BOARD == BOARD_S3
    // ledDebug();

    batt_loop();

    s = getWebValue();
    bool do_dhcp = false;

    if(digitalRead(CONF_PIN) || s.dhcpSwitch){
      do_dhcp = true;
      s.dhcpSwitch = false;
    }
    if (do_dhcp) {
      addLog("Doing DHCP request!", "info");
      Ethernet.begin(ethMac);

      IPAddress ipNone = IPAddress(0,0,0,0);
      bool ipReceived = false;
      uint32_t now = millis();
      while (millis() - now < 10000) {
        delay(1);

        IPAddress got = Ethernet.localIP();
        if (got != ipNone) {
          localIP = got.toString();
          addLog("Got DHCP IP: " + localIP, "ok");
          stringToIPAddress(localIP+"", ethIp);
          ipReceived = true;
          ip_sent = false;   // start sending IP to WWW
          write_settings();
          ethOk = eth_init();

          addLog("Ethernet initiliazing... "+ ethOk, "info");
          s.ipAddress = localIP;
          break;
        }
      }

      if (!ipReceived) addLog("Failed to get DHCP IP!", "err");

      // wait 
      confMode = false;
    }
    
    
    if(s.ipAddress != localIP && s.ipAddress != ""){
      localIP = s.ipAddress;
      addLog("New local IP: " + localIP, "ok");
      stringToIPAddress(localIP+"", ethIp);
      ip_sent = false;   // start sending IP to WWW
      write_settings();
      ethOk = eth_init();

      addLog("Ethernet initiliazing... " + ethOk, "info");
    }
    if(s.batt_switch != isBatteryConnected){
      isBatteryConnected = s.batt_switch;
      d.batt_switch = isBatteryConnected;
      
      write_settings();
      if(isBatteryConnected){
        addLog("Battery mode - ON", "info");
      }else{
        addLog("Battery mode - OFF", "info");
      }
    }

    if(isBatteryConnected){
      if((int)s.real_batt != (int)batteryVoltage && s.real_batt != 0.0f){
        batt_cof = s.real_batt * batt_cof / batteryVoltage;
        write_settings();
        addLog("Battery calibrated - " + String(s.real_batt), "ok");
      }
    }
    //Serial.println(isBatteryConnected);

    d.ipAddress = localIP;
    d.battEsp = String(batteryVoltage, 2);
    initValue(d);
  #endif
}
